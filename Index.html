    <button onclick="checkAuth()">点击提交</button>

    <div id="result">等待结果...</div>

    <script>
        async function checkAuth() {
            const idNo = document.getElementById('idNo').value;
            const name = document.getElementById('name').value;
            const resDiv = document.getElementById('result');

            resDiv.innerText = "正在查询中...";

            try {
                const response = await fetch('https://idenauthen.market.alicloudapi.com/idenAuthentication', {
                    method: 'POST',
                    headers: {
                        'Authorization': 'APPCODE 88fee821e8904528ae93f7955934fa28',
                        'Content-Type': 'application/x-www-form-urlencoded; charset=UTF-8'
                    },
                    body: `idNo=${idNo}&name=${encodeURIComponent(name)}`
                });

                const data = await response.text();
                resDiv.innerText = "返回结果：\n" + data;
            } catch (error) {
                resDiv.innerText = "错误：可能是跨域(CORS)限制或网络问题。\n" + error;
            }
        }
    </script>
